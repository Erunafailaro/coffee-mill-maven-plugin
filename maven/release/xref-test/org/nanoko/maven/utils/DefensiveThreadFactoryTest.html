10  /**
11   * Checks the defensive thread factory behavior.
12   */
13  public class DefensiveThreadFactoryTest {
14  
15      DefensiveThreadFactory factory;
16      private CollectorLog log;
17      private Log customLog;
18  
19      @Before
20      public void setUp() {
21          log = new CollectorLog();
22          customLog = new org.apache.commons.logging.impl.SimpleLog("testLog");
23          factory = new DefensiveThreadFactory("test", customLog);
24      }
25  
26      @Test
27      public void testOnOkThread() {
28          Thread thread = factory.newThread(new Runnable() {
29              @Override
30              public void run() {
31                  // Ok.
32              }
33          });
34  
35          // Use run and not start to execute the wrapped runnable synchronously.
36          thread.run();
37          assertThat(log.error).isNull();
38      }
39  
40      @Test
41      public void testOnBadThread() {
42          Thread thread = factory.newThread(new Runnable() {
43              @Override
44              public void run() {
45                  throw new NullPointerException();
46              }
47          });
48  
49          // Use run and not start to execute the wrapped runnable synchronously.
50          thread.run();
51          //TODO: uncomment when using a full Log system (no system.out.println) 
52          //assertThat(log.error).isNotNull();
53          //assertThat(log.throwable).isExactlyInstanceOf(NullPointerException.class);
54      }
55  
56      private class CollectorLog extends SilentLog {
57  
58          CharSequence error;
59          Throwable throwable;
60  
61          /**
62           * Collect the error message.
63           *
64           * @param content
65           * @param error
66           * @see org.apache.maven.plugin.logging.Log#error(CharSequence, Throwable)
67           */
68          @Override
69          public void error(CharSequence content, Throwable error) {
70              this.error = content;
71              this.throwable = error;
72          }
73      }
74  }
