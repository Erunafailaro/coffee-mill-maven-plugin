12  /**
13   * Aggregate JavaScript sources:
14   * <p/>
15   * Aggregation order can be specified in the Maven Configuration
16   * It can also include:
17   * <ul>
18   * <li>The Javascript file from the project ((<tt>filename.js</tt>)</li>
19   * <li>Dependencies files (<tt>artifactId.js</tt>)</li>
20   * <li>Compile files (CoffeeScript) (<tt>coffee_filename.js</tt>)</li>
21   * </ul>
22   * <p/>
23   * If no file are included, the project js files are aggregated in the alphabetical order.
24   *
25   * @goal aggregate-javascript
26   */
27  public class JavaScriptAggregatorMojo extends AbstractCoffeeMillMojo {
28  
29      /**
30       * @parameter
31       */
32      protected List<String> javascriptAggregation;
33  
34      public void execute() throws MojoExecutionException {
35          // Do we have js files ?
36          if (FileUtils.listFiles(getWorkDirectory(), new String[] {"js"}, true).size() == 0) {
37              getLog().info("Skipping JavaScript aggregation - no files");
38              return;
39          }
40  
41          File output = new File(getWorkDirectory(), project.getBuild().getFinalName() + ".js");
42  
43          JavaScriptAggregator aggregator = new JavaScriptAggregator();
44          Map<String, Object> options = new HashMap<String, Object>();
45          options.put("output", output);
46          options.put("names", javascriptAggregation);
47          options.put("extension", "js");
48          aggregator.configure(this, options);
49          try {
50              aggregator.processAll();
51          } catch (Processor.ProcessorException e) {
52              throw new MojoExecutionException("Cannot aggregate JavaScript files", e);
53          }
54  
55          if (output.isFile()) {
56              try {
57                  File artifact = new File(getTarget(), project.getBuild().getFinalName() + ".js");
58                  getLog().info("Copying " + output.getAbsolutePath() + " to the " + artifact.getAbsolutePath());
59                  FileUtils.copyFile(output, artifact, true);
60                  project.getArtifact().setFile(artifact);
61              } catch (IOException e) {
62                  throw new MojoExecutionException("Cannot copy the aggregated file to the target folder", e);
63              }
64          } else {
65              throw new MojoExecutionException("Cannot copy the aggregated file to the target folder - the output file " +
66                      "does not exist");
67          }
68      }

70  }
