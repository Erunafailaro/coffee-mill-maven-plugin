81      /**
82       * Iterates over project HTML and HTM files and compress them.
83       *
84       * @throws org.nanoko.coffee.mill.processors.Processor.ProcessorException
85       */
86      @Override
87      public void processAll() throws ProcessorException {
88          if (! mojo.assetsDir.exists()) {
89              return;
90          }
91  
92          Collection<File> files = FileUtils.listFiles(mojo.assetsDir, new String[]{"html", "htm"}, true);
93          for (File file : files) {
94              compress(file);
95          }
96      }

114     private void compress(File file) throws ProcessorException {
115         HtmlCompressor htmlCompressor = new HtmlCompressor();
116 
117         htmlCompressor.setCompressCss(false);
118         htmlCompressor.setCompressJavaScript(false);
119         htmlCompressor.setEnabled(true);
120         htmlCompressor.setGenerateStatistics(true);
121         htmlCompressor.setPreserveLineBreaks(preserveLineBreak);
122         htmlCompressor.setPreservePatterns(preservePatterns);
123         htmlCompressor.setRemoveComments(removeComments);
124         htmlCompressor.setRemoveMultiSpaces(removeMultispaces);
125         htmlCompressor.setRemoveFormAttributes(removeFormAttributes);
126         htmlCompressor.setRemoveHttpProtocol(removeHttpProtocol);
127         htmlCompressor.setRemoveHttpsProtocol(removeHttpsProtocol);
128         htmlCompressor.setRemoveInputAttributes(removeInputAttributes);
129         htmlCompressor.setRemoveIntertagSpaces(removeIntertagSpaces);
130         htmlCompressor.setRemoveJavaScriptProtocol(removeJavascriptProtocol);
131         htmlCompressor.setRemoveLinkAttributes(removeLinkAttributes);
132         htmlCompressor.setRemoveQuotes(removeQuotes);
133         htmlCompressor.setRemoveScriptAttributes(removeScriptAttributes);
134         htmlCompressor.setRemoveStyleAttributes(removeStyleAttributes);
135         htmlCompressor.setSimpleBooleanAttributes(simpleBooleanAttributes);
136         htmlCompressor.setSimpleDoctype(simpleDocType);
137 
138         try {
139             String result = htmlCompressor.compress(FileUtils.readFileToString(file));
140             File out = getOutputHtmlFile(file);
141             out.getParentFile().mkdirs();
142             FileUtils.write(out, result);
143             writeStatistics(htmlCompressor, file);
144         } catch(Exception e) {
145             throw new ProcessorException(e.getMessage());
146         }
147 
148 
149         getLog().info("HTML compression completed.");
150     }

152     private void writeStatistics(HtmlCompressor htmlCompressor, File file) {
153         boolean si = true;
154 
155         int origFilesizeBytes = htmlCompressor.getStatistics().getOriginalMetrics().getFilesize();
156         String origFilesize = humanReadableByteCount(origFilesizeBytes, si);
157         String origEmptyChars = String.valueOf(htmlCompressor.getStatistics().getOriginalMetrics().getEmptyChars());
158         String origInlineEventSize = humanReadableByteCount(htmlCompressor.getStatistics().getOriginalMetrics().getInlineEventSize(), si);
159         String origInlineScriptSize =humanReadableByteCount(htmlCompressor.getStatistics().getOriginalMetrics().getInlineScriptSize(), si);
160         String origInlineStyleSize =humanReadableByteCount(htmlCompressor.getStatistics().getOriginalMetrics().getInlineStyleSize(), si);
161 
162         int compFilesizeBytes = htmlCompressor.getStatistics().getCompressedMetrics().getFilesize();
163         String compFilesize =humanReadableByteCount(compFilesizeBytes, si);
164         String compEmptyChars = String.valueOf(htmlCompressor.getStatistics().getCompressedMetrics().getEmptyChars());
165         String compInlineEventSize =humanReadableByteCount(htmlCompressor.getStatistics().getCompressedMetrics().getInlineEventSize(), si);
166         String compInlineScriptSize =humanReadableByteCount(htmlCompressor.getStatistics().getCompressedMetrics().getInlineScriptSize(), si);
167         String compInlineStyleSize =humanReadableByteCount(htmlCompressor.getStatistics().getCompressedMetrics().getInlineStyleSize(), si);
168 
169         String elapsedTime = getElapsedHMSTime(htmlCompressor.getStatistics().getTime());
170         String preservedSize =humanReadableByteCount(htmlCompressor.getStatistics().getPreservedSize(), si);
171         Float compressionRatio = new Float(compFilesizeBytes) / new Float(origFilesizeBytes);
172         Float spaceSavings = new Float(1) - compressionRatio;
173 
174         String format = "%-30s%-30s%-30s%-2s";
175         NumberFormat formatter = new DecimalFormat("#0.00");
176         String eol = "\n";
177         String hr = "+-----------------------------+-----------------------------+-----------------------------+";
178         StringBuilder sb = new StringBuilder(file.getName() + " - HTML compression statistics:").append(eol);
179         sb.append(hr).append(eol);
180         sb.append(String.format(format, "| Category", "| Original", "| Compressed", "|")).append(eol);
181         sb.append(hr).append(eol);
182         sb.append(String.format(format, "| Filesize", "| " + origFilesize, "| " + compFilesize, "|")).append(eol);
183         sb.append(String.format(format, "| Empty Chars", "| " + origEmptyChars, "| " + compEmptyChars, "|")).append(eol);
184         sb.append(String.format(format, "| Script Size", "| " + origInlineScriptSize, "| " + compInlineScriptSize, "|")).append(eol);
185         sb.append(String.format(format, "| Style Size", "| " + origInlineStyleSize, "| " + compInlineStyleSize, "|")).append(eol);
186         sb.append(String.format(format, "| Event Handler Size", "| " + origInlineEventSize, "| " + compInlineEventSize, "|")).append(eol);
187         sb.append(hr).append(eol);
188         sb.append(String.format("%-90s%-2s",
189                 String.format("| Time: %s, Preserved: %s, Compression Ratio: %s, Savings: %s%%",
190                         elapsedTime, preservedSize, formatter.format(compressionRatio), formatter.format(spaceSavings*100)),
191                 "|")).append(eol);
192         sb.append(hr).append(eol);
193 
194         String statistics = sb.toString();
195         getLog().info(statistics);
196     }

198     private File getOutputHtmlFile(File input) {
199         String path = input.getParentFile().getAbsolutePath().substring(mojo.assetsDir.getAbsolutePath().length());
200         return new File(mojo.getWorkDirectory(), path + "/" + input.getName());

203     public static String humanReadableByteCount(long bytes, boolean si) {
204         int unit = si ? 1000 : 1024;
205         if (bytes < unit) return bytes + " B";
206         int exp = (int) (Math.log(bytes) / Math.log(unit));
207         String pre = (si ? "kMGTPE" : "KMGTPE").charAt(exp - 1) + (si ? "" : "i");
208         return String.format("%.1f %sB", bytes / Math.pow(unit, exp), pre);
209     }

211     public static String getElapsedHMSTime(long elapsedTime) {
212         String format = String.format("%%0%dd", 2);
213         elapsedTime = elapsedTime / 1000;
214         String seconds = String.format(format, elapsedTime % 60);
215         String minutes = String.format(format, (elapsedTime % 3600) / 60);
216         String hours = String.format(format, elapsedTime / 3600);
217         return hours + ":" + minutes + ":" + seconds;
218     }
