12  /**
13   * Copy JavaScript sources to the <tt>work</tt> directory and check JavaScript sources with
14   * <ul>
15   *     <li>Check the code using JSLint</li>
16   *     <li>Check the code using JSHint</li>
17   * </ul>
18   *
19   * @goal compile-javascript
20   *
21   */
22  public class JavaScriptCompilerMojo extends AbstractCoffeeMillMojo {
23  
24      /**
25       * Sets to true to disable JSLint
26       *
27       * @parameter default-value="false"
28       */
29      protected boolean skipJsLint;
30  
31      /**
32       * Sets to true to disable JSHint
33       *
34       * @parameter default-value="false"
35       */
36      protected boolean skipJsHint;
37  
38  
39      public void execute() throws MojoExecutionException, MojoFailureException {
40          if (! javaScriptDir.exists()) {
41              getLog().debug("The javascript directory does not exist - skipping JavaScript compilation");
42              return;
43          }
44  
45          JavaScriptFileCopyProcessor processor = new JavaScriptFileCopyProcessor();
46          processor.configure(this, new OptionsHelper.OptionsBuilder().set("test", false).build());
47          try {
48              processor.processAll();
49          } catch (Processor.ProcessorException e) {
50              throw new MojoExecutionException("Cannot copy JavaScript files", e);
51          }
52  
53          if (! skipJsLint) {
54              doJsLint();
55          } else {
56              getLog().debug("JS Lint skipped");
57          }
58  
59          if (! skipJsHint) {
60              doJsHint();
61          } else {
62              getLog().debug("JS Hint skipped");
63          }
64  
65      }
66  
67      private void doJsLint() throws MojoExecutionException {
68          getLog().info("Checking sources with JsLint");
69          JSLintProcessor processor = new JSLintProcessor();
70          processor.configure(this, null);
71          try {
72              processor.processAll();
73          } catch (Processor.ProcessorException e) {
74              throw new MojoExecutionException("", e);
75          }
76      }
77  
78      private void doJsHint() throws MojoExecutionException {
79          getLog().info("Checking sources with JsHint");
80          JSHintProcessor processor = new JSHintProcessor();
81          processor.configure(this, null);
82          try {
83              processor.processAll();
84          } catch (Processor.ProcessorException e) {
85              throw new MojoExecutionException("", e);
86          }
87      }
