16  package org.nanoko.coffee.mill.mojos.compile;
17  
18  import org.apache.maven.plugin.MojoExecutionException;
19  import org.apache.maven.plugin.MojoFailureException;
20  import org.nanoko.coffee.mill.mojos.AbstractCoffeeMillMojo;
21  import org.nanoko.coffee.mill.processors.OptiPNGProcessor;
22  import org.nanoko.coffee.mill.processors.Processor;
23  import org.nanoko.coffee.mill.utils.OptionsHelper;
24  
25  /**
26   * Optimizes PNG files using optiPNG (http://optipng.sourceforge.net/).
27   * OptiPNG must be installed and the executable `optipng` available form the system path.
28   * @goal optimize-png
29   *
30   */
31  public class OptiPNGMojo extends AbstractCoffeeMillMojo {
32  
33      OptiPNGProcessor processor;
34  
35      /**
36       * Enables the verbose mode of optiPNG.
37       * @parameter default-value=true
38       */
39      public boolean optiPNGVerbose;
40  
41      /**
42       * The optiPNG optimization level among 0-7.
43       * @parameter default-value=2
44       */
45      protected int optiPngOptimizationLevel;
46  
47      /**
48       * Skips the PNG files optimization.
49       * @parameter default-value=false
50       */
51      protected boolean skipOptiPNG;
52  
53      public OptiPNGMojo() {
54          processor = new OptiPNGProcessor();
55      }
56  
57      public void execute() throws MojoExecutionException, MojoFailureException {
58          if (skipOptiPNG) {
59              return;
60          }
61  
62          if (optiPngOptimizationLevel < 0  || optiPngOptimizationLevel > 7) {
63              throw new MojoExecutionException("Invalid optiPNG optimization level, it must be in [0-7]");
64          }
65  
66  
67          processor.configure(this, new OptionsHelper.OptionsBuilder().set("verbose", true).set("level",
68                  optiPngOptimizationLevel).build());
69  
70          try {
71              processor.processAll();
72          } catch (Processor.ProcessorException e) {
73              throw new MojoFailureException("PNG Optimization failed", e);
74          }
75      }
76  
77  }
