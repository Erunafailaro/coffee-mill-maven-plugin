18  import org.apache.commons.io.FileUtils;
19  import org.apache.maven.plugin.MojoExecutionException;
20  import org.apache.maven.plugin.MojoFailureException;
21  import org.junit.After;
22  import org.junit.Before;
23  import org.junit.Test;
24  import org.nanoko.coffeemill.mojos.processresources.CopyAssetsMojo;
25  import org.nanoko.coffeemill.mojos.processresources.HtmlCompressorMojo;
26  
27  import java.io.File;
28  import java.util.HashMap;
29  import java.util.Map;
30  
31  import static org.junit.Assert.assertTrue;
32  
33  public class HtmlCompressorMojoTest {
34  	
35  	private final File assetsSourceTestDir = new File("src/test/resources/assets");
36  	private final File workDir = new File("target/test/HtmlCompressorMojoTest/www");
37  	private HtmlCompressorMojo mojo;
38  	
39  	public HtmlCompressorMojoTest() throws MojoExecutionException{
40  		CopyAssetsMojo copymojo = new CopyAssetsMojo();
41  		copymojo.setAssetsDir(assetsSourceTestDir);
42  		copymojo.setWorkDirectory(workDir);
43  		copymojo.execute();
44  	}	
45  	
46  	@Before
47  	public void prepareTestDirectory(){
48      	this.mojo = new HtmlCompressorMojo();
49      	this.mojo.setWorkDirectory(this.workDir);
50      	this.mojo.setAssetsDir(assetsSourceTestDir);
51      }
52  	
53  
54      @Test
55      public void testHTMLCompression() throws MojoExecutionException, MojoFailureException {
56      	System.out.println("\n ==> Should compress the html test file (smaller file size).");
57      	
58      	mojo.setSkipHtmlCompression(false);
59      	
60      	Map<String,String> options = new HashMap<>();
61      	
62      	options.put("preserveLineBreak", "false");
63      	options.put("generateStatistics", "true");
64      	options.put("removeComments", "true");
65      	options.put("removeFormAttributes", "true");
66      	options.put("removeHttpProtocol", "true");
67      	options.put("removeHttpsProtocol", "true");
68      	options.put("removeInputAttributes", "true");
69      	options.put("removeIntertagSpaces", "true");
70      	options.put("removeJavaScriptProtocol", "true");
71      	options.put("removeLinkAttributes", "true");
72      	options.put("removeMultiSpaces", "true");
73      	options.put("removeQuotes", "true");
74      	options.put("removeScriptAttributes", "true");
75      	options.put("removeStyleAttributes", "true");
76      	options.put("simpleBooleanAttributes", "true");
77      	options.put("simpleDoctype", "true");
78      	
79      	mojo.setHtmlCompressionOptions(options); 
80  
81          File file = new File(mojo.getAssetsDir(), "lemonde/le-monde.html");
82          long size = file.length();
83  
84          mojo.execute();
85  
86          file = new File(mojo.getWorkDirectory(), "lemonde/le-monde.html");
87          long newSize = file.length();
88  
89          // Optimization, so the new size is smaller.
90          assertTrue(newSize < size);
91      }
92      
93      @After
94  	public void cleanTestDirectory() {
95  		if(this.mojo.getWorkDirectory().exists()){
96  			FileUtils.deleteQuietly(this.mojo.getWorkDirectory());
97  		}
98  	}
