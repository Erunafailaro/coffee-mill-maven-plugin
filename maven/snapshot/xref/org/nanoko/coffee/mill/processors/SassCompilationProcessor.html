33  /**
34   * Processor handling Sass and Compass to CSS compilation.
35   * It handles <tt>.scss</tt> files from the <tt>stylesheets</tt> directory.
36   */
37  public class SassCompilationProcessor extends DefaultProcessor {
38  
39  
40      protected Map<String, String> sassOptions = new HashMap<String, String>(ImmutableMap.of(
41              "unix_newlines", "true",
42              "cache", "true",
43              "always_update", "true",
44              "style", ":expanded"));
45      /**
46       * Enable the use of Compass style library mixins, this emulates the
47       * {@code --compass} commandline option of Sass.
48       */
49      protected boolean useCompass;
50      private File source;
51      private File destination;
52      private StringBuilder script;
53      private ScriptingContainer jruby;
54      private File frameworks;
55  
56      public void tearDown() {
57          // Do nothing.
58      }

60      @Override
61      public void configure(AbstractCoffeeMillMojo mojo, Map<String, Object> options) {
62          super.configure(mojo, options);
63          this.source = mojo.stylesheetsDir;
64          this.destination = mojo.getWorkDirectory();
65          this.useCompass = OptionsHelper.getBoolean(options, "useCompass", true);
66          this.frameworks = OptionsHelper.getFile(options, "frameworks");
67  
68          this.script = new StringBuilder();
69          buildSASSScript(script);
70          getLog().debug(script.toString());
71  
72          System.setProperty("org.jruby.embed.localcontext.scope", "threadsafe");
73  
74  //        final ScriptEngineManager scriptEngineManager = new ScriptEngineManager();
75  //        this.jruby = scriptEngineManager.getEngineByName("jruby");
76  //        jruby.getContext().setAttribute("org.jruby.embed.sharing.variables", false, ScriptContext.ENGINE_SCOPE);
77  
78          jruby = new ScriptingContainer(LocalContextScope.THREADSAFE, LocalVariableBehavior.GLOBAL);
79  
80      }
81  
82      public boolean accept(File file) {
83          return isFileContainedInDirectory(file, source)
84                  && (file.getName().endsWith(".scss") || file.getName().endsWith(".sass"))
85                  && file.isFile();
86      }
87  
88      @Override
89      public void processAll() throws ProcessorException {
90          if (!source.exists()) {
91              return;
92          }
93          Collection<File> files = FileUtils.listFiles(source, new String[]{"scss", "sass"}, true);
94          if (! files.isEmpty()) {
95              compile();
96          }
97      }
98  
99      private File getOutputCSSFile(File input) {
100         String cssFileName = input.getName().substring(0, input.getName().length() - ".scss".length()) + ".css";
101         String path = input.getParentFile().getAbsolutePath().substring(source.getAbsolutePath().length());
102         return new File(destination, path + "/" + cssFileName);
103     }
104 
105     private void compile() throws ProcessorException {
106         getLog().info("Compiling SASS Templates");
107         //Execute the SASS Compilation Ruby Script
108         final SassCompilationErrors compilationErrors = (SassCompilationErrors) jruby.runScriptlet(script.toString());
109         if (compilationErrors.hasErrors()) {
110             for (SassCompilationErrors.CompilationError error : compilationErrors) {
111                 getLog().error("Compilation of template " + error.filename + " failed: " + error.message);
112             }
113             throw new ProcessorException("SASS compilation encountered errors (see above for details).");
114         }
115     }
116 
117     @Override
118     public void fileCreated(File file) throws ProcessorException {
119         compile();
120     }
121 
122     @Override
123     public void fileUpdated(File file) throws ProcessorException {
124         compile();
125     }

127     @Override
128     public void fileDeleted(File file) {
129         File theFile = getOutputCSSFile(file);
130         if (theFile.exists()) {
131             theFile.delete();
132         }
133     }

135     /**
136      * Dump the Ruby script used to execute the Sass compilation.
137      *
138      * @param script the string builder where the script is written.
139      */
140     protected void buildSASSScript(final StringBuilder script) {
141         script.append("require 'rubygems'\n");
142         script.append("require 'sass/plugin'\n");
143         script.append("require 'java'\n");
144 
145         if (this.useCompass) {
146             getLog().info("Running with Compass enabled.");
147             script.append("require 'compass'\n");
148             script.append("require 'compass/exec'\n");
149             script.append("Compass.add_project_configuration \n");
150             this.sassOptions.put("load_paths", "Compass.configuration.sass_load_paths");
151             // manually specify these paths
152             script.append("Compass::Frameworks.register_directory('jar:'+ File.join(Compass.base_directory, 'frameworks/compass'))\n");
153             script.append("Compass::Frameworks.register_directory('jar:'+ File.join(Compass.base_directory, 'frameworks/blueprint'))\n");
154             script.append("Compass::Frameworks.register_directory('").append(FilenameUtils.separatorsToUnix(frameworks
155                     .getAbsolutePath() + "/compass")).append("')").append("\n");
156             script.append("Compass::Frameworks.register_directory('").append(FilenameUtils.separatorsToUnix(frameworks
157                     .getAbsolutePath() + "/blueprint")).append("')").append("\n");
158         }
159 
160         script.append("Sass::Plugin.options.merge!(\n");
161 
162         //If not explicitly set place the cache location in the target dir
163         if (!this.sassOptions.containsKey("cache_location")) {
164             final File sassCacheDir = new File(destination, "sass_cache");
165             final String sassCacheDirStr = sassCacheDir.toString();
166             this.sassOptions.put("cache_location", "'" + FilenameUtils.separatorsToUnix(sassCacheDirStr) + "'");
167         }
168 
169         //Add the plugin configuration options
170         for (final Iterator<Map.Entry<String, String>> entryItr = this.sassOptions.entrySet().iterator(); entryItr.hasNext(); ) {
171             final Map.Entry<String, String> optEntry = entryItr.next();
172             final String opt = optEntry.getKey();
173             final String value = optEntry.getValue();
174             script.append("    :").append(opt).append(" => ").append(value);
175             if (entryItr.hasNext()) {
176                 script.append(",");
177             }
178             script.append("\n");
179         }
180         script.append(")\n");
181 
182         // set up compilation error reporting
183         script.append("java_import ");
184         script.append(SassCompilationErrors.class.getName());
185         script.append("\n");
186         script.append("$compilation_errors = SassCompilationErrors.new\n");
187         script.append("Sass::Plugin.on_compilation_error {|error, template, css| $compilation_errors.add(template, error.message) }\n");
188 
189         //Add the SASS template locations
190         getLog().info("Queuing SASS Template for compile: " + source.getAbsoluteFile());
191 
192         script.append("Sass::Plugin.add_template_location('")
193                 .append(FilenameUtils.separatorsToUnix(source.getAbsolutePath()))
194                 .append("', '")
195                 .append(FilenameUtils.separatorsToUnix(destination.getAbsolutePath()))
196                 .append("')\n");
197         // make ruby give use some debugging info when requested
198         if (getLog().isDebugEnabled()) {
199             script.append("require 'pp'\npp Sass::Plugin.options\n");
200             script.append("pp Compass::configuration\n");
201         }
202 
203         script.append("Sass::Plugin.update_stylesheets\n");
204         script.append("return $compilation_errors\n");
205     }
