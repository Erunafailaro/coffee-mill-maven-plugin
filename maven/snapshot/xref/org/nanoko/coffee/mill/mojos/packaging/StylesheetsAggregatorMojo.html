15  /**
16   * Aggregate stylesheets sources:
17   *
18   * Aggregation order can be specified in the Maven Configuration
19   * It can also include:
20   * <ul>
21   *     <li>The CSS file from the project ((<tt>filename.js</tt>)</li>
22   *     <li>Compiled files (LESS) (<tt>less_filename.js</tt>)</li>
23   * </ul>
24   *
25   * If no file are included, the project CSS files are aggregated in the alphabetical order.
26   *
27   * @goal aggregate-stylesheets
28   */
29  public class StylesheetsAggregatorMojo extends AbstractCoffeeMillMojo {
30  
31      /**
32       * @parameter
33       */
34      protected List<String> cssAggregation;
35  
36  
37      public void execute() throws MojoExecutionException, MojoFailureException {
38  
39          // Do we have css files ?
40          System.out.println(getWorkDirectory().getAbsolutePath());
41          if (FileUtils.listFiles(getWorkDirectory(), new String[] {"css"}, true).size() == 0) {
42              getLog().info("Skipping Stylessheets aggregation - no files");
43              return;
44          }
45  
46          File output = new File(getWorkDirectory(), project.getBuild().getFinalName() + ".css");
47  
48          Processor aggregator = new CSSAggregator();
49          Map<String, Object> options = new HashMap<String, Object>();
50          options.put("output", output);
51          options.put("names", cssAggregation);
52          options.put("extension", "css");
53          aggregator.configure(this, options);
54          try {
55              aggregator.processAll();
56          } catch (Processor.ProcessorException e) {
57              throw new MojoExecutionException("Cannot aggregate CSS files", e);
58          }
59  
60          if (output.isFile()) {
61              try {
62                  FileUtils.copyFileToDirectory(output, getTarget());
63                  // Do we already have a main JS artifact ?
64                  if (project.getArtifact().getFile() != null  && project.getArtifact().getFile().exists()) {
65                      projectHelper.attachArtifact(project, "css", output);
66                  } else {
67                      project.getArtifact().setFile(output);
68                  }
69              } catch (IOException e) {
70                  throw new MojoExecutionException("Cannot copy the aggregated file to the target folder", e);
71              }
72          } else {
73              throw new MojoExecutionException("Cannot copy the aggregated file to the target folder - the output file " +
74                      "does not exist");
75          }
76  
77      }
78  }
