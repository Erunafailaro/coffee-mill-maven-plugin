127     @SuppressWarnings("unchecked")
128 	public static void extendJasmineMojoForIT(AbstractCoffeeMillMojo mill, AbstractJasmineMojo mojo,
129                                                String reportName) {
130         InjectionHelper.inject(mojo, AbstractJasmineMojo.class, "jasmineTargetDir",
131                 getJasmineITDirectory(mill.project));
132         if (reportName != null) {
133             InjectionHelper.inject(mojo, AbstractJasmineMojo.class, "junitXmlReportFileName",
134                     reportName);
135         }
136 
137         List<String> deps = new ArrayList<String>();
138         for (Dependency dep : (Collection<Dependency>) mill.project.getDependencies()) {
139             if ("js".equals(dep.getType())) {
140                 String filename = dep.getArtifactId() + ".js";
141                 if (dep.getClassifier() != null  && ! dep.getClassifier().equals("min")) {
142                     filename = dep.getArtifactId() + "-" + dep.getClassifier() + ".js";
143                 }
144                 File file = new File(mill.getLibDirectory(), filename);
145 
146                 if (! file.exists()) {
147                     mill.getLog().error("Cannot preload " + dep.getArtifactId() + ":" + dep.getVersion() + " : " +
148                             file
149                                     .getAbsolutePath() + " not found");
150                 } else {
151                     try {
152                         FileUtils.copyFileToDirectory(file, getJasmineITDirectory(mill.project));
153                     } catch (IOException e) {
154                     	mill.getLog().error("Error during copy files to Jasmine directory "+getJasmineITDirectory(mill.project).getAbsolutePath(), e);
155                     }
156                     deps.add(filename);
157                 }
158             }
159         }
160     }

162     public static void configureJasmineToRunOnLibrary(AbstractJasmineMojo mojo, String library) {
163         if (library != null) {
164             List<String> list = new ArrayList<String>();
165             list.add(library);
166             InjectionHelper.inject(mojo, AbstractJasmineMojo.class, "sourceIncludes",
167                     list);
168         }

195     public static void copyJunitReport(AbstractCoffeeMillMojo millMojo, File report, String classname) {
196         // Copy the resulting junit report if exit
197         if (report.isFile()) {
198             try {
199                 String reportContent = FileUtils.readFileToString(report);
200                 if (classname != null) {
201                     reportContent = reportContent.replace("classname=\"jasmine\"", "classname=\"" + classname + "\"");
202                 }
203                 File surefire = new File(millMojo.project.getBuild().getDirectory(), "surefire-reports");
204                 surefire.mkdirs();
205                 File newReport = new File(surefire, report.getName());
206                 FileUtils.write(newReport, reportContent);
207             } catch (IOException e) {
208                 millMojo.getLog().error("Cannot write the surefire report", e);
209             }
210         }
211     }
