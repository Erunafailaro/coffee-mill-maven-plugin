20  public class CSSCompilerMojo extends AbstractCoffeeMillMojo {
21  
22      /**
23       * Sets to true to disable CSSLint
24       *
25       * @parameter default-value="false"
26       */
27      protected boolean skipCSSLint;
28  
29  
30      public void execute() throws MojoExecutionException, MojoFailureException {
31          if (! stylesheetsDir.exists()) {
32              getLog().info("The stylesheet directory does not exist - skipping CSS compilation");
33              return;
34          }
35  
36          CSSFileCopyProcessor processor = new CSSFileCopyProcessor();
37          processor.configure(this, null);
38          try {
39              processor.processAll();
40          } catch (Processor.ProcessorException e) {
41              throw new MojoExecutionException("Cannot copy CSS files", e);
42          }
43  
44          if (! skipCSSLint) {
45              lint();
46          }
47      }
48  
49      private void lint() throws MojoFailureException {
50          CSSLintProcessor processor = new CSSLintProcessor();
51          processor.configure(this, new OptionsHelper.OptionsBuilder().set("directory", getWorkDirectory()).build());
52  
53          Collection<File> files = FileUtils.listFiles(getWorkDirectory(), new String[]{"css"}, true);
54          for (File file : files) {
55              try {
56                  processor.processAll();
57              } catch (Processor.ProcessorException e) {
58                  getLog().error("Cannot run the CSS Lint Processor on " + file.getAbsolutePath(), e);
59              }
60          }
61      }
