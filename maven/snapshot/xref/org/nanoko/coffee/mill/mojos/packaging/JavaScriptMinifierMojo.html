89      public void execute() throws MojoExecutionException, MojoFailureException {
90          if (skipMinification) {
91              getLog().debug("JavaScript Minification skipped");
92              return;
93  
94          }
95          if (! project.getArtifact().getFile().exists()  || ! project.getArtifact().getFile().getName().endsWith("" +
96                  ".js")) {
97              throw new MojoExecutionException("Cannot minify the project artifact - either the file does not exist or " +
98                      "is not a JavaScript file");
99          }
100 
101         File output = new File(getTarget(), project.getBuild().getFinalName() + "-min.js");
102 
103         if (Minifier.GOOGLE_CLOSURE.equals(minifier)) {
104             doGoogleCompression(project.getArtifact().getFile(), output);
105         } else if (Minifier.YUI_COMPRESSOR.equals(minifier)) {
106             doYUICompression(project.getArtifact().getFile(), output);
107         }
108 
109         if (! output.isFile()) {
110             throw new MojoFailureException("The minified file "+ output.getAbsolutePath() + " does not exist - check " +
111                     "log.");
112         }
113 
114         if (attachMinifiedJavaScript) {
115             projectHelper.attachArtifact(project, output, "min");
116         }
117     }

119     private void doGoogleCompression(File file, File output) throws MojoExecutionException {
120         getLog().info("Compressing " + file.getName() + " using Google Closure");
121         final com.google.javascript.jscomp.Compiler compiler = new Compiler();
122         CompilerOptions options = newCompilerOptions();
123         getLog().info("Compilation Level set to " + minifierGoogleCompilationLevel);
124         minifierGoogleCompilationLevel.setOptionsForCompilationLevel(options);
125 
126         final JSSourceFile[] input = new JSSourceFile[] {
127                 JSSourceFile.fromFile(file)
128         };
129         JSSourceFile[] externs = new JSSourceFile[] {};
130 
131         compiler.initOptions(options);
132         final Result result = compiler.compile(externs, input, options);
133         if (result.success) {
134             try {
135                 FileUtils.write(output, compiler.toSource());
136             } catch (IOException e) {
137                 throw new MojoExecutionException("Cannot write minified file", e);
138             }
139         } else {
140             for (JSError error : result.errors) {
141                 getLog().error(error.sourceName + ":" + error.lineNumber + " - " + error.description);
142             }
143             throw new MojoExecutionException("The minification failed - check log");
144         }
145 
146     }

148     private void doYUICompression(File file, File output) throws MojoExecutionException {
149         getLog().info("Compressing " + file.getName() + " using YUI Compressor");
150         YUIErrorReporter reporter = new YUIErrorReporter();
151         try {
152             final JavaScriptCompressor compressor = new JavaScriptCompressor(new FileReader(file),
153                 reporter);
154             FileWriter writer = new FileWriter(output);
155             int linebreakpos = -1;
156             compressor.compress(writer, linebreakpos, minifierYUIMunge, minifierYUIVerbose, minifierYUIPreserveSemiColumn,
157                     minifierYUIDisableOptimizations);
158             writer.close();
159             if (reporter.errorFound) {
160                 throw new MojoExecutionException("Error during minification - Check log");
161             }
162         } catch (IOException e) {
163             throw new MojoExecutionException("Error during minification", e);
164         }
165     }

168     /**
169      * @return default {@link CompilerOptions} object to be used by compressor.
170      */
171     protected CompilerOptions newCompilerOptions() {
172         final CompilerOptions options = new CompilerOptions();
173         /**
174          * According to John Lenz from the Closure Compiler project, if you are using the Compiler API directly, you
175          * should specify a CodingConvention. {@link http://code.google.com/p/wro4j/issues/detail?id=155}
176          */
177         options.setCodingConvention(new ClosureCodingConvention());
178         //set it to warning, otherwise compiler will fail
179         options.setWarningLevel(DiagnosticGroups.CHECK_VARIABLES,
180                 CheckLevel.WARNING);
181         return options;
182     }
