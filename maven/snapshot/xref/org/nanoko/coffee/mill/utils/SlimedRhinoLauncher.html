45          Context context = Context.enter();
46          context.setOptimizationLevel(9);

75      public Object evaluate(final String script, final String sourceName){
76      	Context context = Context.enter();
77      	context.setOptimizationLevel(9);
78          Object result = context.evaluateString(compileScope, script, sourceName, 0, null);
79          Context.exit();
80          return result;
81      }
82      
83      public Object evaluate(final InputStream stream, final String sourceName) throws IOException{
84      	Context context = Context.enter();
85          Object result = context.evaluateReader(compileScope, new InputStreamReader(stream), sourceName, 0, null);
86          Context.exit();
87          return result;
88      }

96      public static String toJSMultiLineString(String data){
97          final String[] lines = data.split("\n");
98          final StringBuffer result = new StringBuffer("[");
99          if (lines.length == 0) {
100             result.append("\"\"");
101         }
102         for (int i = 0; i < lines.length; i++) {
103             final String line = lines[i];
104             result.append("\"");
105             result.append(line.replace("\\", "\\\\").replace("\"", "\\\"").replaceAll("\\r|\\n", ""));
106             // this is used to force a single line to have at least one new line (otherwise cssLint fails).
107             if (lines.length == 1) {
108                 result.append("\\n");
109             }
110             result.append("\"");
111             if (i < lines.length - 1) {
112                 result.append(",");
113             }
114         }
115         result.append("].join(\"\\n\")");
116         return result.toString();
117     }
