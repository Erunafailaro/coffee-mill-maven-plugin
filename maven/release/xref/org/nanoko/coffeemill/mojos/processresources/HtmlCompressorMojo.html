93      public void execute() throws MojoExecutionException {
94          if(isSkipped()) { 
95              return; 
96          }
97  
98          this.configure();
99  
100         try {
101             Collection<File> files = FileUtils.listFiles(getAssetsDir(), new String[]{"html", "htm"}, true);
102             for (File file : files) {          
103                 compress(file);
104             }        
105         } catch (WatchingException e) {
106             throw new MojoExecutionException("Error during execute() on HtmlCompressorMojo", e);
107         }        
108     }

131     private boolean compress(File file) throws WatchingException {
132         getLog().info("Compress Html file "+file.getName() +" from " + this.getAssetsDir().getAbsolutePath());
133         try {
134             String result = htmlCompressor.compress(FileUtils.readFileToString(file));
135             File out = FSUtils.computeRelativeFile(file, getAssetsDir(), getWorkDirectory());
136             if(out.exists()){
137                 FileUtils.deleteQuietly(out);
138             }
139             out.getParentFile().mkdirs();
140             FileUtils.write(out, result);
141             writeStatistics(htmlCompressor, file);
142         } catch(IOException e) {
143             throw new WatchingException("Error during Html compression on file "+file.getName(), e);
144         }
145 
146         getLog().info("HTML compression completed.");
147         return true;
148     }

243     private void writeStatistics(HtmlCompressor htmlCompressor, File file) {
244         boolean si = true;
245 
246         int origFilesizeBytes = htmlCompressor.getStatistics().getOriginalMetrics().getFilesize();
247         String origFilesize = humanReadableByteCount(origFilesizeBytes, si);
248         String origEmptyChars = String.valueOf(htmlCompressor.getStatistics().getOriginalMetrics().getEmptyChars());
249         String origInlineEventSize = humanReadableByteCount(htmlCompressor.getStatistics().getOriginalMetrics().getInlineEventSize(), si);
250         String origInlineScriptSize =humanReadableByteCount(htmlCompressor.getStatistics().getOriginalMetrics().getInlineScriptSize(), si);
251         String origInlineStyleSize =humanReadableByteCount(htmlCompressor.getStatistics().getOriginalMetrics().getInlineStyleSize(), si);
252 
253         int compFilesizeBytes = htmlCompressor.getStatistics().getCompressedMetrics().getFilesize();
254         String compFilesize =humanReadableByteCount(compFilesizeBytes, si);
255         String compEmptyChars = String.valueOf(htmlCompressor.getStatistics().getCompressedMetrics().getEmptyChars());
256         String compInlineEventSize =humanReadableByteCount(htmlCompressor.getStatistics().getCompressedMetrics().getInlineEventSize(), si);
257         String compInlineScriptSize =humanReadableByteCount(htmlCompressor.getStatistics().getCompressedMetrics().getInlineScriptSize(), si);
258         String compInlineStyleSize =humanReadableByteCount(htmlCompressor.getStatistics().getCompressedMetrics().getInlineStyleSize(), si);
259 
260         String elapsedTime = getElapsedHMSTime(htmlCompressor.getStatistics().getTime());
261         String preservedSize =humanReadableByteCount(htmlCompressor.getStatistics().getPreservedSize(), si);
262         Float compressionRatio = new Float(compFilesizeBytes) / new Float(origFilesizeBytes);
263         Float spaceSavings = new Float(1) - compressionRatio;
264 
265         String format = "%-30s%-30s%-30s%-2s";
266         NumberFormat formatter = new DecimalFormat("#0.00");
267         String eol = "\n";
268         String hr = "+-----------------------------+-----------------------------+-----------------------------+";
269 
270         StringBuilder sb2 = new StringBuilder().append(eol);
271         sb2.append(file.getName() + " - HTML compression statistics:").append(eol);
272         sb2.append(hr).append(eol);
273         sb2.append(String.format(format, "| Category", "| Original", "| Compressed", "|")).append(eol);
274         sb2.append(hr).append(eol);
275         sb2.append(String.format(format, "| Filesize", "| " + origFilesize, "| " + compFilesize, "|")).append(eol);
276         sb2.append(String.format(format, "| Empty Chars", "| " + origEmptyChars, "| " + compEmptyChars, "|")).append(eol);
277         sb2.append(String.format(format, "| Script Size", "| " + origInlineScriptSize, "| " + compInlineScriptSize, "|")).append(eol);
278         sb2.append(String.format(format, "| Style Size", "| " + origInlineStyleSize, "| " + compInlineStyleSize, "|")).append(eol);
279         sb2.append(String.format(format, "| Event Handler Size", "| " + origInlineEventSize, "| " + compInlineEventSize, "|")).append(eol);
280         sb2.append(hr).append(eol);
281         sb2.append(
282                 String.format("%-90s%-2s",
283                         String.format("| Time: %s, Preserved: %s, Compression Ratio: %s, Savings: %s%%",
284                                 elapsedTime, preservedSize, formatter.format(compressionRatio), formatter.format(spaceSavings*100)),
285                         "|")).append(eol);
286         sb2.append(hr).append(eol);
287         String statistics = sb2.toString();
288         getLog().info(statistics);
289     }

291     private static String humanReadableByteCount(long bytes, boolean si) {
292         int unit = si ? 1000 : 1024;
293         if (bytes < unit) { 
294             return bytes + " B"; 
295         }
296         int exp = (int) (Math.log(bytes) / Math.log(unit));
297         String pre = (si ? "kMGTPE" : "KMGTPE").charAt(exp - 1) + (si ? "" : "i");
298         return String.format("%.1f %sB", bytes / Math.pow(unit, exp), pre);
299     }

301     private static String getElapsedHMSTime(long elapsedTime) {
302         String format = String.format("%%0%dd", 2);
303         long newElapsedTime = elapsedTime / 1000;
304         String seconds = String.format(format, newElapsedTime % 60);
305         String minutes = String.format(format, (newElapsedTime % 3600) / 60);
306         String hours = String.format(format, newElapsedTime / 3600);
307         return hours + ":" + minutes + ":" + seconds;
308     }
