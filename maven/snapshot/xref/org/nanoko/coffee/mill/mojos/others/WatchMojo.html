61  /**
62   * This mojo watches the file change in the source directories and process them automatically.
63   * To work correctly, launch <tt>mvn clean test</tt> first. This will resolve and prepare all required file.
64   * Then <tt>mvn org.nanoko.coffee-mill:coffee-mill-maven-plugin:watch</tt> will starts the <i>watch</i> mode.
65   * <p/>
66   * This mojo supports reactor mode, i.e. is able to watch several modules and updates files. To enable this mode,
67   * launch the watch mode with <tt>-Dwatched.project=artifactId of the final project</tt>. This will watch all
68   * resources of all the modules of the reactor and copy the resulting artifact on the other module in the specified
69   * project.
70   * <p/>
71   * You can configure the watched port with the <tt>-Dwatch.port=8234</tt> option. By default the used port is 8234.
72   *
73   * @goal watch
74   */
75  public class WatchMojo extends AbstractCoffeeMillMojo implements FileListener {

161     public void execute() throws MojoExecutionException, MojoFailureException {
162         // Are we in reactor mode, if so are we the target project
163         if (!watchedProject.equals(project.getArtifactId())) {
164             getLog().debug("Not the watched project, skip");
165             if ("js".equals(project.getPackaging())) {
166                 getLog().debug("Adding the current project to session " + watchedProject);
167                 ReactorUtils.addWatcherToSession(this, session);
168             }
169             return;
170         }
171 
172         processors = new ArrayList<Processor>();
173         computeProcessors(this, processors);
174 
175 
176         try {
177             setupMonitor(project);
178         } catch (FileSystemException e) {
179             throw new MojoExecutionException("Cannot set the file monitor on the source folder", e);
180         }
181 
182         // Starts all others process and monitors on the others project
183         for (WatchMojo watcher : ReactorUtils.getWatchersFromSession(session)) {
184             computeProcessors(watcher, processors);
185             DefaultProcessor proc = new FinalArtifactProcessor(this, watcher);
186             proc.configure(this, null);
187             processors.add(proc);
188             try {
189                 setupMonitor(watcher.project);
190             } catch (FileSystemException e) {
191                 throw new MojoExecutionException("Cannot set the file monitor on the source folder", e);
192             }
193         }
194 
195         String MESSAGE = "You're running the watch mode. All modified files will be processed " +
196                 "automatically. \n" +
197                 "If the jetty server is enabled, they will also be served from http://localhost:" +
198                 watchJettyServerPort + "/. \n" +
199                 "The jasmine runner is available from http://localhost:" + watchJettyServerPort + "/jasmine. \n" +
200                 "To leave the watch mode, just hit CTRL+C.\n";
201         getLog().info(MESSAGE);
202 
203         for (Processor processor : processors) {
204             try {
205                 processor.processAll();
206             } catch (Processor.ProcessorException e) {
207                 getLog().error("", e);
208             }
209         }
210 
211         if (watchRunServer) {
212             try {
213                 server = new Server();
214                 addConnectorToServer();
215                 addHandlersToServer();
216                 startServer();
217             } catch (Exception e) {
218                 throw new MojoExecutionException("Cannot run the jetty server", e);
219             }
220         } else {
221             try {
222                 Thread.sleep(1000000000); // Pretty long
223             } catch (InterruptedException e) { /* ignore */ }
224         }
225     }

227     private List<Processor> computeProcessors(WatchMojo mojo, List<Processor> processors) {
228         // Always added
229 
230         // Asset Copy
231         DefaultProcessor processor = new CopyAssetProcessor();
232         processor.configure(mojo, null);
233         processors.add(processor);
234 
235         // Copy JS Main + Test
236         processor = new JavaScriptFileCopyProcessor();
237         processor.configure(mojo, new OptionsHelper.OptionsBuilder().set("test", false).build());
238         processors.add(processor);
239         processor = new JavaScriptFileCopyProcessor();
240         processor.configure(mojo, new OptionsHelper.OptionsBuilder().set("test",
241                 true).build());
242         processors.add(processor);
243 
244         // Copy CSS
245         processor = new CSSFileCopyProcessor();
246         processor.configure(mojo, null);
247         processors.add(processor);
248 
249         // Less
250         if (watchLess) {
251             processor = new LessCompilationProcessor();
252             processor.configure(mojo, null);
253             processors.add(processor);
254         }
255 
256         // Sass
257         if (watchSass) {
258             try {
259                 File frameworks = SassCompilerMojo.expandCompassFrameworks(this);
260                 processor = new SassCompilationProcessor();
261                 processor.configure(this, new OptionsHelper.OptionsBuilder()
262                         .set("frameworks", frameworks)
263                         .set("useCompass", true) // Compass enabled by default.
264                         .build());
265                 processors.add(processor);
266             } catch (IOException e) {
267                 getLog().error("Cannot expand compass-frameworks, Sass / Compass disabled", e);
268             }
269         }
270 
271         // CoffeeScript
272         if (watchCoffeeScript) {
273             File coffee = CoffeeScriptCompilerMojo.getCoffeeScript(this);
274             if (coffee == null) {
275                 getLog().error("Cannot find the coffeescript compiler, coffeescript disabled");
276             } else {
277                 processor = new CoffeeScriptCompilationProcessor();
278                 processor.configure(mojo, new OptionsHelper.OptionsBuilder()
279                         .set("test", false)
280                         .set("coffeescript", coffee).build());
281                 processors.add(processor);
282 
283                 processor = new CoffeeScriptCompilationProcessor();
284                 processor.configure(mojo, new OptionsHelper.OptionsBuilder()
285                         .set("test", true)
286                         .set("coffeescript", coffee).build());
287                 processors.add(processor);
288             }
289         }
290 
291         if (watchDust) {
292             processor = new DustJSProcessor();
293             processor.configure(mojo, null);
294             processors.add(processor);
295         }
296 
297         // JS and CSS Aggregation
298         if (watchDoAggregate) {
299             processor = new JavaScriptAggregator();
300             Map<String, Object> options = new HashMap<String, Object>();
301             File output = new File(mojo.getWorkDirectory(), mojo.project.getBuild().getFinalName() + ".js");
302             options.put("output", output);
303             options.put("names", mojo.javascriptAggregation);
304             options.put("extension", "js");
305             processor.configure(mojo, options);
306             processors.add(processor);
307 
308             processor = new CSSAggregator();
309             output = new File(mojo.getWorkDirectory(), mojo.project.getBuild().getFinalName() + ".css");
310             options = new HashMap<String, Object>();
311             options.put("output", output);
312             options.put("names", mojo.cssAggregation);
313             options.put("extension", "css");
314             processor.configure(mojo, options);
315             processors.add(processor);
316         }
317 
318         // CSSLint, JSLint and JSHint validation
319         if (watchValidateJS) {
320             processor = new JSHintProcessor();
321             processor.configure(mojo, null);
322             processors.add(processor);
323 
324             processor = new JSHintProcessor();
325             processor.configure(mojo, null);
326             processors.add(processor);
327         }
328         if (watchValidateCSS) {
329             processor = new CSSLintProcessor();
330             processor.configure(mojo, new OptionsHelper.OptionsBuilder().set("directory",
331                     mojo.getWorkDirectory()).build());
332             processors.add(processor);
333         }
334 
335         if (watchOptimizeAssets) {
336             // Asset optimization
337             processor = new OptiPNGProcessor();
338             processor.configure(mojo, new OptionsHelper.OptionsBuilder().set("verbose", true).set("level",
339                     optiPngOptimizationLevel).build());
340 
341             processor = new JpegTranProcessor();
342             processor.configure(this, new OptionsHelper.OptionsBuilder().set("verbose", true).build());
343 
344             // HTML Compression
345             processor = new HTMLCompressorProcessor();
346             Map<String, Object> options = new OptionsHelper.OptionsBuilder()
347                     .set("preserveLineBreak", true)
348                     .set("removeComments", true)
349                     .set("removeMultispaces", true)
350                     .set("removeFormAttributes", true)
351                     .set("removeHttpProtocol", true)
352                     .set("removeHttpsProtocol", true)
353                     .set("removeInputAttributes", true)
354                     .set("removeIntertagSpaces", true)
355                     .set("removeJavascriptProtocol", true)
356                     .set("removeLinkAttributes", true)
357                     .set("removeQuotes", true)
358                     .set("removeScriptAttributes", true)
359                     .set("simpleBooleanAttributes", true)
360                     .set("removeStyleAttributes", true)
361                     .set("simpleDocType", true)
362                     .build();
363             processor.configure(this, options);
364         }
365 
366         return processors;
367     }

369     private void setupMonitor(MavenProject project) throws FileSystemException {
370         File baseDir = project.getBasedir();
371         getLog().info("Set up file monitor on " + baseDir);
372         FileSystemManager fsManager = VFS.getManager();
373         FileObject dir = fsManager.resolveFile(baseDir.getAbsolutePath());
374 
375         DefaultFileMonitor fm = new DefaultFileMonitor(this);
376         fm.setRecursive(true);
377         fm.addFile(dir);
378         fm.start();
379     }

401     public void fileCreated(FileChangeEvent event) throws Exception {
402         getLog().info("New file found " + event.getFile().getName().getBaseName());
403         boolean processed = false;
404         String path = event.getFile().getName().getPath();
405         File theFile = new File(path);
406         for (Processor processor : processors) {
407             if (processor.accept(theFile)) {
408                 processed = true;
409                 processor.fileCreated(theFile);
410             }
411         }
412 
413         if (!processed) {
414             getLog().info("Nothing to do for " + event.getFile().getName().getBaseName());
415         }
416     }
