23  /**
24   * This mojo watches the file change in the source directories and process them automatically.
25   * To work correctly, launch <tt>mvn clean test</tt> first. This will resolve and prepare all required file.
26   * Then <tt>mvn org.nano.coffee-mill:coffee-mill-maven-plugin:watch</tt> will starts the <i>watch</i> mode.
27   *
28   * This mojo supports reactor mode, i.e. is able to watch several modules and updates files. To enable this mode,
29   * launch the watch mode with <tt>-Dwatched.project=artifactId of the final project</tt>. This will watch all
30   * resources of all the modules of the reactor and copy the resulting artifact on the other module in the specified
31   * project.
32   * @goal watch
33   */
34  public class WatchMojo extends AbstractCoffeeMillMojo implements FileListener {

131     public void execute() throws MojoExecutionException, MojoFailureException {
132         // Are we in reactor mode, if so are we the target project
133         if (! watchedProject.equals(project.getArtifactId())) {
134             getLog().debug("Not the watched project, skip");
135             if ("js".equals(project.getPackaging())) {
136                 getLog().debug("Adding the current project to session " + watchedProject);
137                 ReactorUtils.addWatcherToSession(this, session);
138             }
139             return;
140         }
141 
142         processors = new ArrayList<Processor>();
143         computeProcessors(this, processors);
144 
145 
146         try {
147             setupMonitor(project);
148         } catch (FileSystemException e) {
149             throw new MojoExecutionException("Cannot set the file monitor on the source folder", e);
150         }
151 
152         // Starts all others process and monitors on the others project
153         for (WatchMojo watcher : ReactorUtils.getWatchersFromSession(session)) {
154             computeProcessors(watcher, processors);
155             DefaultProcessor proc = new FinalArtifactProcessor(this, watcher);
156             proc.configure(this, null);
157             processors.add(proc);
158             try {
159                 setupMonitor(watcher.project);
160             } catch (FileSystemException e) {
161                 throw new MojoExecutionException("Cannot set the file monitor on the source folder", e);
162             }
163         }
164 
165         String MESSAGE = "You're running the watch mode. All modified files will be processed " +
166                 "automatically. \n" +
167                 "If the jetty server is enabled, they will also be served from http://localhost:" +
168                 watchJettyServerPort + "/. \n" +
169                 "The jasmine runner is available from http://localhost:" + watchJettyServerPort + "/jasmine. \n" +
170                 "To leave the watch mode, just hit CTRL+C.\n";
171         getLog().info(MESSAGE);
172 
173         for (Processor processor : processors) {
174             try {
175                 processor.processAll();
176             } catch (Processor.ProcessorException e) {
177                 getLog().error("", e);
178             }
179         }
180 
181         if (watchRunServer) {
182             try {
183                 server = new Server();
184                 addConnectorToServer();
185                 addHandlersToServer();
186                 startServer();
187             } catch (Exception e){
188                 throw new MojoExecutionException("Cannot run the jetty server", e);
189             }
190         } else {
191             try {
192                 Thread.sleep(1000000000); // Pretty long
193             } catch (InterruptedException e) { /* ignore */ }
194         }
195     }

197     private List<Processor> computeProcessors(WatchMojo mojo, List<Processor> processors) {
198         // Always added
199 
200         // Asset Copy
201         DefaultProcessor processor = new CopyAssetProcessor();
202         processor.configure(mojo, null);
203         processors.add(processor);
204 
205         // Copy JS Main + Test
206         processor = new JavaScriptFileCopyProcessor();
207         processor.configure(mojo, new OptionsHelper.OptionsBuilder().set("test", false).build());
208         processors.add(processor);
209         processor = new JavaScriptFileCopyProcessor();
210         processor.configure(mojo, new OptionsHelper.OptionsBuilder().set("test",
211                 true).build());
212         processors.add(processor);
213 
214         // Copy CSS
215         processor = new CSSFileCopyProcessor();
216         processor.configure(mojo, null);
217         processors.add(processor);
218 
219         // Less
220         if (watchLess) {
221             processor = new LessCompilationProcessor();
222             processor.configure(mojo, null);
223             processors.add(processor);
224         }
225 
226         // CoffeeScript
227         if (watchCoffeeScript) {
228             processor = new CoffeeScriptCompilationProcessor();
229             processor.configure(mojo, new OptionsHelper.OptionsBuilder().set("test",
230                     false).build());
231             processors.add(processor);
232 
233             processor = new CoffeeScriptCompilationProcessor();
234             processor.configure(mojo, new OptionsHelper.OptionsBuilder().set("test",
235                     true).build());
236             processors.add(processor);
237         }
238 
239         if (watchDust) {
240             processor = new DustJSProcessor();
241             processor.configure(mojo, null);
242             processors.add(processor);
243         }
244 
245         // JS and CSS Aggregation
246         if (watchDoAggregate) {
247             processor = new JavaScriptAggregator();
248             Map<String, Object> options = new HashMap<String, Object>();
249             File output = new File(mojo.getWorkDirectory(), mojo.project.getBuild().getFinalName() + ".js");
250             options.put("output", output);
251             options.put("names", mojo.javascriptAggregation);
252             options.put("extension", "js");
253             processor.configure(mojo, options);
254             processors.add(processor);
255 
256             processor = new CSSAggregator();
257             output = new File(mojo.getWorkDirectory(), mojo.project.getBuild().getFinalName() + ".css");
258             options = new HashMap<String, Object>();
259             options.put("output", output);
260             options.put("names", mojo.cssAggregation);
261             options.put("extension", "css");
262             processor.configure(mojo, options);
263             processors.add(processor);
264         }
265 
266         // CSSLint, JSLint and JSHint validation
267         if (watchValidateJS) {
268             processor = new JSHintProcessor();
269             processor.configure(mojo, null);
270             processors.add(processor);
271 
272             processor = new JSHintProcessor();
273             processor.configure(mojo, null);
274             processors.add(processor);
275         }
276         if (watchValidateCSS) {
277             processor = new CSSLintProcessor();
278             processor.configure(mojo, new OptionsHelper.OptionsBuilder().set("directory",
279                     mojo.getWorkDirectory()).build());
280             processors.add(processor);
281         }
282 
283         if (watchOptimizeAssets) {
284             // Asset optimization
285             processor = new OptiPNGProcessor();
286             processor.configure(mojo, new OptionsHelper.OptionsBuilder().set("verbose", true).set("level",
287                     optiPngOptimizationLevel).build());
288 
289             processor = new JpegTranProcessor();
290             processor.configure(this, new OptionsHelper.OptionsBuilder().set("verbose", true).build());
291         }
292 
293         return processors;
294     }

297     private void setupMonitor(MavenProject project) throws FileSystemException {
298         File baseDir = project.getBasedir();
299         getLog().info("Set up file monitor on " + baseDir);
300         FileSystemManager fsManager = VFS.getManager();
301         FileObject dir = fsManager.resolveFile(baseDir.getAbsolutePath());
302 
303         DefaultFileMonitor fm = new DefaultFileMonitor(this);
304         fm.setRecursive(true);
305         fm.addFile(dir);
306         fm.start();
307     }

329     public void fileCreated(FileChangeEvent event) throws Exception {
330         getLog().info("New file found " + event.getFile().getName().getBaseName());
331         boolean processed = false;
332         String path = event.getFile().getName().getPath();
333         File theFile = new File(path);
334         for (Processor processor : processors) {
335             if (processor.accept(theFile)) {
336                 processed = true;
337                 processor.fileCreated(theFile);
338             }
339         }
340 
341         if (! processed) {
342             getLog().info("Nothing to do for " + event.getFile().getName().getBaseName());
343         }
344     }
