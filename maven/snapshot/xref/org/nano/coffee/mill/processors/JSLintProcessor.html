15   * Validates a JS file using JSHint
16   */
17  public class JSLintProcessor extends DefaultProcessor {
18  
19      @Override
20      public void processAll() throws ProcessorException {
21          getLog().info("Checking sources with JsLint");
22          Collection<File> files = FileUtils.listFiles(millMojo.getWorkDirectory(), new String[]{"js"}, true);
23          for (File file : files) {
24              List<ProcessorWarning> warnings = validate(file);
25              getLog().info("Found " + warnings.size() + " issues in " + file.getAbsolutePath());
26              for (ProcessorWarning warning: warnings) {
27                  getLog().warn("In " + warning.file.getName() + " @" + warning.line + ":" + warning.character
28                          + " -> " + warning.evidence + " - " + warning.reason);
29              }
30          }
31      }
32  
33      @Override
34      public boolean accept(File file) {
35          return file.isFile() && isFileContainedInDirectory(file, millMojo.getWorkDirectory());
36      }
37  
38      @Override
39      public void fileCreated(File file) throws ProcessorException {
40          validate(file);
41      }
42  
43      @Override
44      public void fileUpdated(File file) throws ProcessorException {
45          validate(file);
46      }

48      public List<ProcessorWarning> validate(File file) {
49          List<ProcessorWarning> warnings = new ArrayList<ProcessorWarning>();
50          JsLint jslint = new JsLint();
51          getLog().debug("JSLint-ing " + file.getAbsolutePath());
52          try {
53              jslint.validate(FileUtils.readFileToString(file));
54          } catch (IOException e) {
55              getLog().error("Can't analyze " + file.getAbsolutePath() + " with JSLint", e);
56          } catch (LinterException e) {
57              if (!e.getErrors().isEmpty()) {
58                  for (LinterError exp : e.getErrors()) {
59                      if (exp == null) {
60                          continue;
61                      }
62                      ProcessorWarning warning = new ProcessorWarning(file, exp.getLine(), exp.getCharacter(),
63                              exp.getEvidence(), exp.getReason());
64                      warnings.add(warning);
65                  }
66              }
67          }
68          return warnings;
69      }
